        <label class="file-upload-label" for="pdf-upload">上传 Zine (PDF)</label>

        <input type="file" id="pdf-upload" accept="application/pdf" />

        

        <button id="prev-page" disabled>◀ 上一页</button>

        <span class="page-info">

            <span id="page-num">0</span> / <span id="page-count">0</span>

        </span>

        <button id="next-page" disabled>下一页 ▶</button>

    </div>


    <div class="canvas-container">

        <canvas id="pdf-render"></canvas>

    </div>


    <script>

        // 配置 PDF.js 的 worker 路径

        const pdfjsLib = window['pdfjs-dist/build/pdf'];

        pdfjsLib.GlobalWorkerOptions.workerSrc = 'https://cdnjs.cloudflare.com/ajax/libs/pdf.js/2.16.105/pdf.worker.min.js';


        let pdfDoc = null,

            pageNum = 1,

            pageIsRendering = false,

            pageNumIsPending = null;


        const scale = 1.5; // 控制渲染清晰度，数值越大越清晰但也越耗内存

        const canvas = document.querySelector('#pdf-render');

        const ctx = canvas.getContext('2d');


        // 渲染指定页面

        const renderPage = num => {

            pageIsRendering = true;


            // 获取页面

            pdfDoc.getPage(num).then(page => {

                const viewport = page.getViewport({ scale });

                canvas.height = viewport.height;

                canvas.width = viewport.width;


                const renderCtx = {

                    canvasContext: ctx,

                    viewport: viewport

                };


                page.render(renderCtx).promise.then(() => {

                    pageIsRendering = false;


                    if (pageNumIsPending !== null) {

                        renderPage(pageNumIsPending);

                        pageNumIsPending = null;

                    }

                });


                // 更新当前页码显示

                document.querySelector('#page-num').textContent = num;

                

                // 更新按钮状态

                document.querySelector('#prev-page').disabled = num <= 1;

                document.querySelector('#next-page').disabled = num >= pdfDoc.numPages;

            });

        };


        // 检查是否有待处理的渲染任务

        const queueRenderPage = num => {

            if (pageIsRendering) {

                pageNumIsPending = num;

            } else {

                renderPage(num);

            }

        };


        // 上一页

        const showPrevPage = () => {

            if (pageNum <= 1) return;

            pageNum--;

            queueRenderPage(pageNum);

        };


        // 下一页

        const showNextPage = () => {

            if (pageNum >= pdfDoc.numPages) return;

            pageNum++;

            queueRenderPage(pageNum);

        };


        // 处理 PDF 上传

        document.querySelector('#pdf-upload').addEventListener('change', function(e) {

            const file = e.target.files[0];

            if (file.type !== 'application/pdf') {

                alert('请上传 PDF 格式的文件！');

                return;

            }


            const fileReader = new FileReader();

            fileReader.onload = function() {

                const typedarray = new Uint8Array(this.result);


                // 加载 PDF 文档

                pdfjsLib.getDocument(typedarray).promise.then(pdfDoc_ => {

                    pdfDoc = pdfDoc_;

                    document.querySelector('#page-count').textContent = pdfDoc.numPages;

                    

                    // 重置页码为1并渲染

                    pageNum = 1;

                    renderPage(pageNum);

                }).catch(err => {

                    console.error("PDF 加载错误: ", err);

                    alert("加载 PDF 时出现错误。");

                });

            };

            fileReader.readAsArrayBuffer(file);

        });


        // 绑定按钮事件

        document.querySelector('#prev-page').addEventListener('click', showPrevPage);

        document.querySelector('#next-page').addEventListener('click', showNextPage);

    </script>
